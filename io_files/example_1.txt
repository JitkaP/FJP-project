const int SIZE := 5;
var int array[SIZE];

procedure fillArray;
var int i;
begin
    i := 0;
    while (i < SIZE) do
        begin
            array[i] := i;
            i := i+1;
        end
end;

procedure countSum;
var int i, sum;
begin
    sum := 0;
    for i := 0 to SIZE do
        sum := (sum + array[i]);
end;

begin
    call fillArray;
    call countSum;
end
